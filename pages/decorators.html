What are Decorators?

✓

Real-Life: Decorators are like stickers on a laptop 💻. The laptop works the same, but stickers add personality (extra behavior). A @Log decorator adds logging without changing the original method.

!
Functions that modify classes/methods at definition time
@ Syntax
// A decorator is just a function!
// It runs when the class is DEFINED, not when methods are called

// Enable in tsconfig.json:
// "experimentalDecorators": true

// Basic usage:
@MyDecorator
class MyClass { }

// On methods:
class API {
  @Log
  fetchData() { ... }
}
Basics

Class Decorators

1
Modify or replace the class itself
@Class
// A class decorator receives the constructor
function Sealed(constructor: Function) {
  Object.seal(constructor);
  Object.seal(constructor.prototype);
  console.log(`${constructor.name} is sealed!`);
}

@Sealed
class User {
  name: string;
  constructor(name: string) { this.name = name; }
}
// Logs: "User is sealed!" — at class definition time

// Decorator factory — decorator that takes arguments
function Component(config: { selector: string }) {
  return function(constructor: Function) {
    console.log(`Component ${config.selector} registered`);
  };
}

@Component({ selector: "app-header" })
class HeaderComponent { }
// Logs: "Component app-header registered"
Basics

Method Decorators

2
Wrap or modify method behavior
@Method
// Method decorator receives: target, name, descriptor
function Log(
  target: any,
  propertyKey: string,
  descriptor: PropertyDescriptor
) {
  const original = descriptor.value;

  descriptor.value = function(...args: any[]) {
    console.log(`Calling ${propertyKey} with`, args);
    const result = original.apply(this, args);
    console.log(`${propertyKey} returned`, result);
    return result;
  };
}

class Calculator {
  @Log
  add(a: number, b: number): number {
    return a + b;
  }
}

const calc = new Calculator();
calc.add(2, 3);
// Calling add with [2, 3]
// add returned 5

Property Decorators

3
Add metadata to properties
@Property
// Property decorator — track which fields are required
const requiredFields: string[] = [];

function Required(target: any, propertyKey: string) {
  requiredFields.push(propertyKey);
}

class Form {
  @Required name: string = "";
  @Required email: string = "";
  phone: string = "";  // optional, no decorator
}

console.log(requiredFields);  // ["name", "email"]
// Now you can validate: check all required fields are filled

Real-World: Logging Decorator

!
Add logging without modifying business logic
Logging
function Measure(target: any, key: string, desc: PropertyDescriptor) {
  const original = desc.value;
  desc.value = function(...args: any[]) {
    const start = performance.now();
    const result = original.apply(this, args);
    const end = performance.now();
    console.log(`⏱ ${key} took ${(end - start).toFixed(2)}ms`);
    return result;
  };
}

class DataService {
  @Measure
  processLargeDataset(data: number[]) {
    return data.sort((a, b) => a - b);
  }
}
// ⏱ processLargeDataset took 12.34ms

Real-World: Memoization

!
Cache expensive computations
Memo
function Memoize(target: any, key: string, desc: PropertyDescriptor) {
  const original = desc.value;
  const cache = new Map<string, any>();

  desc.value = function(...args: any[]) {
    const cacheKey = JSON.stringify(args);
    if (cache.has(cacheKey)) {
      console.log(`Cache hit for ${key}(${cacheKey})`);
      return cache.get(cacheKey);
    }
    const result = original.apply(this, args);
    cache.set(cacheKey, result);
    return result;
  };
}

class MathService {
  @Memoize
  fibonacci(n: number): number {
    if (n <= 1) return n;
    return this.fibonacci(n - 1) + this.fibonacci(n - 2);
  }
}
Watch Out!

⚠ Common Mistakes

⚠
Mistake 1: Forgetting experimentalDecorators 
// Add to tsconfig.json:
// "experimentalDecorators": true
// Without this, @ syntax won't work!
⚠
Mistake 2: Decorator execution order 
// Multiple decorators run BOTTOM to TOP (closest to class first)
@A
@B
@C
class X { }
// Execution order: C → B → A (bottom-up!)
⚠
Mistake 3: Decorators run at DEFINITION, not at runtime 
// Class decorators run when the class is loaded,
// NOT when instances are created!
// Method decorators wrap the method at definition time.

Cheatsheet

// CLASS — receives constructor
function Dec(constructor: Function) { }

// METHOD — receives target, name, descriptor
function Dec(t: any, key: string, desc: PropertyDescriptor) { }

// PROPERTY — receives target, name
function Dec(target: any, key: string) { }

// FACTORY — decorator that takes args
function Dec(args) { return function(target) { }; }

// ORDER: bottom-up for multiple decorators
// ENABLE: "experimentalDecorators": true in tsconfig.json
i

TS 5.0+ Update: TypeScript 5.0 introduced native ECMAScript decorators (Stage 3 proposal) — no experimentalDecorators flag needed. The syntax is nearly identical but uses the new TC39 standard. If you're on TS 5.0+, you can use decorators without the tsconfig flag. Both APIs coexist — the flag controls which spec you target.
